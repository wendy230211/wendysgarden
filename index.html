<script>

function search() {

const email = document.getElementById("email").value.trim().toLowerCase();

const username = document.getElementById("username").value.trim();


if (!email || !username) {

alert("請同時輸入電子郵件與帳號名稱");

return;

}


document.getElementById("result").innerHTML = "查詢中，請稍候...";


fetch("https://script.google.com/macros/s/AKfycbxxxxxxxxxxxx/exec?email="

+ encodeURIComponent(email)

+ "&username="

+ encodeURIComponent(username))

.then(res => res.json())

.then(res => {

const div = document.getElementById("result");


if (!res.success || res.count === 0) {

div.innerHTML = "查無資料";

return;

}


let html = "<table><tr>";

res.header.forEach(h => html += `<th>${h}</th>`);

html += "</tr>";


res.data.forEach(row => {

html += "<tr>";

row.forEach(cell => html += `<td>${cell}</td>`);

html += "</tr>";

});


html += "</table>";

div.innerHTML = html;

})

.catch(() => {

document.getElementById("result").innerHTML = "系統錯誤，請稍後再試";

});

}

</script>
